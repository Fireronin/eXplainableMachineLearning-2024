def calc_accuracy(y_true, y_pred):
    return sum(y_true == y_pred)/len(y_true)


def eval_logistic_regression(df_train_x, df_test_x, df_train_y, df_test_y):
    model = LogisticRegression()
    model.fit(df_train_x, df_train_y)
    train_score = model.score(df_train_x, df_train_y)
    test_score = model.score(df_test_x, df_test_y)

    return train_score, test_score


def eval_random_forest(df_train_x, df_test_x, df_train_y, df_test_y):
    model = RandomForestRegressor()
    model.fit(df_train_x, df_train_y)
    train_score = model.score(df_train_x, df_train_y)
    test_score = model.score(df_test_x, df_test_y)

    return train_score, test_score


def eval_xgboost(df_train_x, df_test_x, df_train_y, df_test_y):
    model = XGBClassifier(use_label_encoder=False, eval_metric='mlogloss')
    model.fit(df_train_x, df_train_y)
    y_train_pred = model.predict(df_train_x)
    y_test_pred = model.predict(df_test_x)
    return (calc_accuracy(df_train_y.to_numpy().flatten(), y_train_pred),
            calc_accuracy(df_test_y.to_numpy().flatten(), y_test_pred))
